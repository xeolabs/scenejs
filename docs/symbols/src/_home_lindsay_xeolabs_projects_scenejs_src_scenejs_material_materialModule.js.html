1 /**
  2  * Backend that manages the current material properties.
  3  *
  4  * Services the SceneJS.material scene node, providing it with methods to set and get the current material.
  5  *
  6  * Interacts with the shading backend through events; on a SHADER_RENDERING event it will respond with a
  7  * MATERIAL_EXPORTED to pass the material properties to the shading backend.
  8  *
  9  * Avoids redundant export of the material properties with a dirty flag; they are only exported when that is set, which
 10  * occurs when material is set by the SceneJS.material node, or on SCENE_ACTIVATED, SHADER_ACTIVATED and
 11  * SHADER_DEACTIVATED events.
 12  *
 13  * Sets the properties to defaults on SCENE_ACTIVATED.
 14  *
 15  * Whenever a SceneJS.material sets the material properties, this backend publishes it with a MATERIAL_UPDATED to allow
 16  * other dependent backends to synchronise their resources. One such backend is the shader backend, which taylors the
 17  * active shader according to the material properties.
 18  *
 19  *  @private
 20  */
 21 var SceneJS_materialModule = new (function() {
 22 
 23     var material;
 24     var dirty;
 25 
 26     SceneJS_eventModule.onEvent(
 27             SceneJS_eventModule.SCENE_ACTIVATED,
 28             function() {
 29                 material = {
 30                     baseColor : [ 0.5, 0.5, 0.5 ],
 31                     specularColor: [ 0.0,  0.0,  0.0 ],
 32                     specular : 0,
 33                     shine : 0,
 34                     reflect : 0,
 35                     alpha : 1.0,
 36                     emit : 0.0
 37                 };
 38                 dirty = true;
 39             });
 40 
 41 
 42     SceneJS_eventModule.onEvent(
 43             SceneJS_eventModule.SHADER_ACTIVATED,
 44             function() {
 45                 dirty = true;
 46             });
 47 
 48     SceneJS_eventModule.onEvent(
 49             SceneJS_eventModule.SHADER_RENDERING,
 50             function() {
 51                 if (dirty) {
 52                     SceneJS_eventModule.fireEvent(
 53                             SceneJS_eventModule.MATERIAL_EXPORTED,
 54                             material);
 55                     dirty = false;
 56                 }
 57             });
 58 
 59     SceneJS_eventModule.onEvent(
 60             SceneJS_eventModule.SHADER_DEACTIVATED,
 61             function() {
 62                 dirty = true;
 63             });
 64 
 65     // @private
 66     this.setMaterial = function(m) {
 67         material = m;
 68         SceneJS_eventModule.fireEvent(
 69                 SceneJS_eventModule.MATERIAL_UPDATED,
 70                 material);
 71         dirty = true;
 72     };
 73 
 74     // @private
 75     this.getMaterial = function() {
 76         return material;
 77     };
 78 })();
