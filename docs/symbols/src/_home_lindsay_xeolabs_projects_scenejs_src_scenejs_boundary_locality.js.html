  4 SceneJS.locality = function() {
  5     var cfg = SceneJS._utils.getNodeConfig(arguments);
  6     var backend = SceneJS._backends.getBackend('view-locality');   
  7 
  8     return SceneJS._utils.createNode(
  9             "locality",
 10             cfg.children,
 11 
 12             new (function() {
 13 
 14                 var _radii = {
 15                     inner : 1000,
 16                     outer : 1500
 17                 };
 18 
 19                 this.setInner = function(inner) {
 20                     _radii.inner = inner;
 21                 };
 22 
 23                 this.getInner = function() {
 24                     return _radii.inner;
 25                 };
 26 
 27                 this.setOuter = function(outer) {
 28                     _radii.outer = outer;
 29                 };
 30 
 31                 this.getOuter = function() {
 32                     return _radii.outer;
 33                 };
 34 
 35                 this._init = function(params) {
 36                     if (params.inner) {
 37                         this.setInner(params.inner);
 38                     }
 39                     if (params.outer) {
 40                         this.setOuter(params.outer);
 41                     }
 42                 };
 43 
 44                 if (cfg.fixed) {
 45                     this._init(cfg.getParams());
 46                 }
 47 
 48                 this._render = function(traversalContext, data) {
 49                     if (!cfg.fixed) {
 50                         _radii.inner = params.inner || 1000;
 51                         _radii.outer = params.outer || _radii.inner + 500.0;
 52                     }
 53                     var prevRadii = backend.getRadii();
 54                     backend.setRadii(_radii);
 55                     SceneJS._utils.visitChildren(cfg, traversalContext, data);
 56                     backend.setRadii(prevRadii);
 57                 };
 58             })());
 59 };
