1 /** Scene node that sets WebGL state for nodes in its subtree.
  2  *
  3  * @class SceneJS.Renderer
  4  * @extends SceneJS.Node
  5  */
  6 SceneJS.Renderer = function() {
  7     SceneJS.Node.apply(this, arguments);
  8     this._nodeType = "renderer";
  9     this._params = this._getParams();
 10     this._sceneId = null; // lazy-set on render
 11     this._lastRenderedData = null;
 12 };
 13 
 14 SceneJS._utils.inherit(SceneJS.Renderer, SceneJS.Node);
 15 
 16 // @private
 17 SceneJS.Renderer.prototype._render = function(traversalContext, data) {
 18     if (this._memoLevel == 0) {
 19         this._rendererState = SceneJS_rendererModule.createRendererState(this._getParams(data));
 20         if (this._fixedParams) {
 21             this._memoLevel = 1;
 22         }
 23     }
 24     SceneJS_rendererModule.setRendererState(this._rendererState);
 25     this._renderNodes(traversalContext, data);
 26     SceneJS_rendererModule.undoRendererState(this._rendererState);
 27 };
 28 
 29 SceneJS.renderer = function() {
 30     var n = new SceneJS.Renderer();
 31     SceneJS.Renderer.prototype.constructor.apply(n, arguments);
 32     return n;
 33 };
