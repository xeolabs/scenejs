1 /**
  2  * Backend module that defines SceneJS events and provides an interface on the backend context through which
  3  * backend modules can fire and subscribe to them.
  4  *
  5  * Events are actually somewhat more like commands; they are always synchronous, and are often used to decouple the
  6  * transfer of data between backends, request events in response, and generally trigger some immediate action.
  7  *
  8  * Event subscription can optionally be prioritised, to control the order in which the subscriber will be notified of
  9  * a given event relative to other suscribers. This is useful, for example, when a backend must be the first to handle
 10  * an INIT, or the last to handle a RESET.
 11  *
 12  */
 13

 78 SceneJS._backends.installBackend(
 79 
 80         "events",
 81 
 82         function(ctx) {
 83 
 84             /* Initialise event types list
 85              */
 86             var nevents = 0;
 87             for (var key in SceneJS._eventTypes) {
 88                 SceneJS._eventTypes[key] = nevents;
 89                 nevents++;
 90             }
 91             var events = new Array(nevents);
 92 
 93             /**
 94              * Registers a handler for the given event
 95              *
 96              * The handler can be registered with an optional priority number which specifies the order it is
 97              * called among the other handler already registered for the event.
 98              *
 99              * So, with n being the number of commands registered for the given event:
100              *
101              * (priority <= 0)      - command will be the first called
102              * (priority >= n)      - command will be the last called
103              * (0 < priority < n)   - command will be called at the order given by the priority
104              *
105              * @param type Event type - one of the values in SceneJS._eventTypes
106              * @param command - Handler function that will accept whatever parameter object accompanies the event
107              * @param priority - Optional priority number (see above)
108              */
109             function onEvent(type, command, priority) {
110                 var list = events[type];
111                 if (!list) {
112                     list = [];
113                     events[type] = list;
114                 }
115                 var handler = {
116                     command: command,
117                     priority : (priority == undefined) ? list.length : priority
118                 };
119                 for (var i = 0; i < list.length; i++) {
120                     if (list[i].priority > handler.priority) {
121                         list.splice(i, 0, handler);
122                         return;
123                     }
124                 }
125                 list.push(handler);
126             }
127 
128             function fireEvent(type, params) {
129                 var list = events[type];
130                 if (list) {
131                     for (var i = 0; i < list.length; i++) {
132                         list[i].command(params || {});
133                     }
134                 }
135             }
136 
137             /* Backend-facing API
138              */
139             ctx.events = {
140                 onEvent : onEvent,
141                 fireEvent : fireEvent
142             };
143 
144             /* Node-facing API            
145              */
146             return {
147                 onEvent : onEvent,
148                 fireEvent : fireEvent
149             };
150         });
