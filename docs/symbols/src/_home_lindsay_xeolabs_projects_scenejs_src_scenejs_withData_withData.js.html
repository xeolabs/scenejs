1 /**
  2  * Scene node that creates a child data containing the elements of its configuration.
  3  */
  4 SceneJS.withData = function() {
  5     var cfg = SceneJS._utils.getNodeConfig(arguments);
  6 
  7     const NO_MEMO = 0;              // No memoization, assuming that node's configuration is dynamic
  8     const FIXED_CONFIG = 1;         // Node config is fixed, memoizing matrix
  9     const FIXED_DATA = 2;         // Node config is fixed, memoizing matrix
 10 
 11     return SceneJS._utils.createNode(
 12             "withData",
 13             cfg.children,
 14 
 15             new (function() {
 16                 var _memoLevel = NO_MEMO;
 17                 var _data = {};
 18                 var _childData;
 19 
 20                 this.setProperty = function(key, value) {
 21                     _data[key] = value;
 22                     _memoLevel = NO_MEMO;
 23                     return this;
 24                 };
 25 
 26                 this.getProperty = function(key) {
 27                     return data[key];
 28                 };
 29 
 30                 this.clearProperties = function() {
 31                     _data = {};
 32                     return this;
 33                 };
 34 
 35                 this._init = function(params) {
 36                     for (var key in params) {
 37                         _data[key] = params[key];
 38                     }
 39                 };
 40 
 41                 if (cfg.fixed) {
 42                     this._init(cfg.getParams());
 43                 }
 44 
 45                 this._render = function(traversalContext, data) {
 46                     if (_memoLevel == NO_MEMO) {
 47                         if (!cfg.fixed) {
 48                             this._init(cfg.getParams(data));
 49                         } else {
 50                             _memoLevel = FIXED_CONFIG;
 51                         }
 52                     }
 53                     if (_memoLevel < FIXED_DATA) {
 54                         _childData = SceneJS._utils.newScope(data, cfg.fixed);
 55                         for (var key in _data) {
 56                             _childData.put(key, _data[key]);
 57                         }
 58                         if (_memoLevel == FIXED_CONFIG && data.isfixed()) {
 59                             _memoLevel = FIXED_DATA;
 60                         }
 61                     }
 62                     this._renderChildren(traversalContext, _childData);
 63                 };
 64             })());
 65 };
