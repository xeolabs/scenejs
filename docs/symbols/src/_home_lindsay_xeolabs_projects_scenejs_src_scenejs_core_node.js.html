1 /** Basic scene graph node, generally used as a group node.
  2  */
  3 SceneJS.node = function() {
  4     var cfg = SceneJS._utils.getNodeConfig(arguments);
  5     return {
  6         _render: function(traversalContext, data) {
  7             var params = cfg.getParams();
  8             var childScope = SceneJS._utils.newScope(data, false);
  9             if (params) {
 10                 for (var key in params) {
 11                     childScope.put(key, params[key]);
 12                 }
 13             }
 14             SceneJS._utils.visitChildren(cfg, traversalContext, childScope || data);
 15         }
 16     };
 17 };
