1 /**
  2  * Backend module that provides the current system time, updating it every time a scene is rendered
  3  *  @private
  4  */
  5 var SceneJS_timeModule = new (function() {
  6 
  7     var time = (new Date()).getTime();
  8 
  9     SceneJS_eventModule.onEvent(
 10             SceneJS_eventModule.SCENE_ACTIVATED,
 11             function() {
 12                 time = (new Date()).getTime();
 13                 SceneJS_eventModule.fireEvent(SceneJS_eventModule.TIME_UPDATED, time);
 14             });
 15 
 16     this.getTime = function() {
 17         return time;
 18     };
 19 })();
